            { id: 5, type: 'tax', bill: 'Delinquency Penalty', fine: 'Up to 40% of tax due', solution: 'Predictive Escrow AI', detail: 'Analyzes gross receipts (e.g. $4.25/$1,000 for parking) and ensures zero late fees.' },
            { id: 6, type: 'labor', bill: 'AB 692 Stay-or-Pay Clauses', fine: '$5,000 per worker', solution: 'Legal Draft AI', detail: 'Automatically flags illegal contract terms in employee agreements to prevent lawsuits.' }
        ];

        let calcChartInstance = null;

        // --- CHART INITIALIZATION ---
        const initCharts = () => {
            const riskCtx = document.getElementById('riskChart').getContext('2d');
            new Chart(riskCtx, {
                type: 'doughnut',
                data: {
                    labels: ['Labor (SB 294/642/AB 692)', 'Tax (BTRC/Delinquency)', 'Safety (Building Code)', 'Environmental (SB 261)'],
                    datasets: [{
                        data: [45, 30, 20, 5],
                        backgroundColor: ['#4f46e5', '#1e40af', '#0ea5e9', '#0f172a'],
                        borderWidth: 0,
                        hoverOffset: 15
                    }]
                },
                options: {
                    responsive: true,
                    maintainAspectRatio: false,
                    cutout: '75%',
                    plugins: {
                        legend: { position: 'bottom', labels: { font: { weight: 'bold', size: 10 }, color: '#64748b' } }
                    }
                }
            });

            // Init Calculator Chart
            const calcCtx = document.getElementById('calcChart').getContext('2d');
            calcChartInstance = new Chart(calcCtx, {
                type: 'bar',
                data: {
                    labels: ['Day 1', 'Day 30', 'Day 60', 'Day 90', 'Day 120'],
                    datasets: [
                        {
                            label: 'Penalty Growth (Manual)',
                            data: [250, 1000, 1500, 3000, 5000],
                            backgroundColor: '#4f46e5',
                            borderRadius: 6
                        },
                        {
                            label: 'Penalty Growth (ITCloudX AI)',
                            data: [50, 50, 50, 50, 50],
                            backgroundColor: '#0ea5e9',
                            borderRadius: 6
                        }
                    ]
                },
                options: {
                    responsive: true,
                    maintainAspectRatio: false,
                    scales: {
                        y: { display: false },
                        x: { grid: { display: false }, ticks: { color: '#94a3b8', font: { size: 10 } } }
                    },
                    plugins: {
                        legend: { position: 'top', labels: { color: '#fff', font: { size: 10 } } }
                    }
                }
            });
        };

        // --- FILTER LOGIC ---
        const renderSolutions = (filter = 'all') => {
            const grid = document.getElementById('solution-grid');
            grid.innerHTML = '';

            const filtered = filter === 'all' ? solutions : solutions.filter(s => s.type === filter);

            filtered.forEach(s => {
                const card = document.createElement('div');
                card.className = 'glass-effect p-8 rounded-3xl space-y-4 hover:border-indigo-400 transition-all duration-300 group section-fade';
                card.innerHTML = `
                    <div class="flex justify-between items-start">
                        <span class="text-xs font-black uppercase tracking-widest px-2 py-1 bg-slate-100 rounded text-slate-500">${s.type}</span>
                        <div class="text-indigo-600 font-black text-xl group-hover:scale-110 transition-transform">✓</div>
                    </div>
                    <div>
                        <h4 class="text-lg font-bold text-slate-900 mb-1">${s.bill}</h4>
                        <p class="text-xs text-red-500 font-bold">Fine: ${s.fine}</p>
                    </div>
                    <div class="p-4 bg-indigo-50 rounded-2xl border border-indigo-100">
                        <div class="text-sm font-bold text-indigo-700">${s.solution}</div>
                        <p class="text-xs text-indigo-600 mt-1">${s.detail}</p>
                    </div>
                `;
                grid.appendChild(card);
            });
        };

        const filterSolutions = (type) => {
            document.querySelectorAll('.filter-btn').forEach(btn => {
                if (btn.getAttribute('data-id') === type) {
                    btn.classList.add('bg-indigo-600', 'text-white', 'border-indigo-600');
                    btn.classList.remove('text-slate-600');
                } else {
                    btn.classList.remove('bg-indigo-600', 'text-white', 'border-indigo-600');
                    btn.classList.add('text-slate-600');
                }
            });
            renderSolutions(type);
        };

        // --- CALCULATOR LOGIC ---
        const calculateSavings = () => {
            const days = parseInt(document.getElementById('delinquencyRange').value);
            document.getElementById('dayDisplay').innerText = `${days} Day${days > 1 ? 's' : ''}`;
            
            // Logic based on BTRC Infraction ($250/day) + escalation
            const baseFine = 250;
            let total = baseFine;
            
            if (days > 1) {
                total = baseFine * days;
            }
            
            // Add tax delinquency logic (5% per month)
            const months = Math.floor(days / 30);
            if (months > 0) {
                total += (total * (0.05 * Math.min(months, 4)));
            }
            if (months > 4) {
                total += (total * 0.20);
            }

            document.getElementById('fineTotal').innerText = new Intl.NumberFormat('en-US', { style: 'currency', currency: 'USD' }).format(total);
            
            // Update Calc Chart Dynamic (Simulation)
            calcChartInstance.data.datasets[0].data = [
                baseFine, 
                baseFine * 30, 
                (baseFine * 60) * 1.05, 
                (baseFine * 90) * 1.10, 
                (baseFine * 120) * 1.40
            ];
            calcChartInstance.update();
        };

        const scrollToSection = (id) => {
            const el = document.getElementById(id);
            if (el) window.scrollTo({ top: el.offsetTop - 100, behavior: 'smooth' });
        };

        // --- INIT ---
        window.onload = () => {
            initCharts();
            renderSolutions('all');
            calculateSavings();
        };
    </script>
